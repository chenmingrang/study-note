
--查看表空间文件
SELECT FILE_NAME,TABLESPACE_NAME FROM DBA_DATA_FILES;

--创建表空间
CREATE TABLESPACE UTF8DB_DATA
LOGGING
DATAFILE 'E:\APP\ADMINISTRATOR\ORADATA\UTF8DB\UTF8DB_DATA.DBF' 
SIZE 32M 
AUTOEXTEND ON 
NEXT 32M MAXSIZE 2048M
EXTENT MANAGEMENT LOCAL;

--创建临时表空间
CREATE TEMPORARY TABLESPACE UTF8DB_TEMP
TEMPFILE 'E:\APP\ADMINISTRATOR\ORADATA\UTF8DB\UTF8DB_TEMP.DBF'
SIZE 32M
AUTOEXTEND ON
NEXT 32M MAXSIZE 2048M
EXTENT MANAGEMENT LOCAL;

--创建用户
DROP USER zzsglnew cascade;
create user zzsglnew identified by zzsgl;
--用户授权
grant dba to zzsglnew;

--修改用户表空间
alter user zzsglnew
DEFAULT TABLESPACE ZZSGL_DATA
TEMPORARY TABLESPACE ZZSGL_temp;
