
--查看表空间文件
SELECT FILE_NAME,TABLESPACE_NAME FROM DBA_DATA_FILES;

--创建表空间
CREATE TABLESPACE WHDF_DATA
LOGGING
DATAFILE '/home/oracle/app/oradata/orcl/WHDF_DATA.DBF' 
SIZE 32M 
AUTOEXTEND ON 
NEXT 32M MAXSIZE 2048M
EXTENT MANAGEMENT LOCAL;

--创建临时表空间
CREATE TEMPORARY TABLESPACE WHDF_TEMP
TEMPFILE '/home/oracle/app/oradata/orcl/WHDF_TEMP.DBF'
SIZE 32M
AUTOEXTEND ON
NEXT 32M MAXSIZE 2048M
EXTENT MANAGEMENT LOCAL;

--创建用户
DROP USER whdf cascade;
create user whdf identified by whdf;
--用户授权
grant dba to whdf;

--修改用户表空间
alter user whdf
DEFAULT TABLESPACE WHDF_DATA
TEMPORARY TABLESPACE WHDF_TEMP;
